Google スプレッドシート連携

Apps Script をデプロイして URL を設定すると、保存時に自動でスプレッドシートの実績列に書き込まれます。

var SHEET_ID = '1EtwwDK4dqKRPdhqFDdDYQ9zDH9hJAk3LuK-1UtpoyGw';
var SHEET_NAME = '202601';

function doPost(e) {
  try {
    var d = JSON.parse(e.postData.contents);
    var sh = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME);
    var row = findRow(sh, d.date);
    if (row === -1) return out('DATE_NOT_FOUND');
    var s = sh, r = row;
    if (d.weight != null) s.getRange(r,12).setValue(d.weight);
    if (d.steps  != null) s.getRange(r,13).setValue(d.steps);
    if (d.burned != null) s.getRange(r,14).setValue(d.burned);
    if (d.intake != null) s.getRange(r,15).setValue(d.intake);
    if (d.breakfast) s.getRange(r,16).setValue(d.breakfast);
    if (d.calB   != null) s.getRange(r,17).setValue(d.calB);
    if (d.lunch)     s.getRange(r,18).setValue(d.lunch);
    if (d.calL   != null) s.getRange(r,19).setValue(d.calL);
    if (d.dinner)    s.getRange(r,20).setValue(d.dinner);
    if (d.calD   != null) s.getRange(r,21).setValue(d.calD);
    if (d.training)  s.getRange(r,22).setValue(d.training);
    return out('OK');
  } catch(err) { return out('ERROR: ' + err.message); }
}

function doGet(e) {
  try {
    var date = e.parameter.date;
    var sh = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME);
    var row = findRow(sh, date);
    if (row === -1) return outJson({});
    var data = sh.getDataRange().getValues();
    var r = data[row - 1];
    return outJson({
      targetWeight: r[2]  || null,
      targetIntake: r[3]  || null,
      targetBurned: r[4]  || null,
      breakfast:    r[6]  || '',
      lunch:        r[7]  || '',
      dinner:       r[8]  || '',
      training:     r[9]  || ''
    });
  } catch(err) { return outJson({ error: err.message }); }
}

function findRow(sh, dateStr) {
  // dateStr = "YYYY/M/D" → ゼロパディングして "YYYY/MM/DD" に正規化
  var parts = dateStr.split('/');
  if (parts.length !== 3) return -1;
  var y = parts[0];
  var m = ('0' + parts[1]).slice(-2);
  var d = ('0' + parts[2]).slice(-2);
  var target = y + '/' + m + '/' + d; // 例: "2026/04/22"

  var ss = sh.getParent();
  var tz = ss.getSpreadsheetTimeZone();
  var data = sh.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    var cd = data[i][0];
    if (!cd) continue;
    // スプシのタイムゾーンで文字列化して比較（タイムゾーンズレを完全回避）
    var cellStr = cd instanceof Date
      ? Utilities.formatDate(cd, tz, 'yyyy/MM/dd')
      : String(cd).trim();
    if (cellStr === target) return i + 1;
  }
  return -1;
}
function out(t){ return ContentService.createTextOutput(t); }
function outJson(o){
  return ContentService.createTextOutput(JSON.stringify(o))
    .setMimeType(ContentService.MimeType.JSON);
}
// ※ GASのWebアプリはリダイレクトが発生するため、
//    アプリ側で redirect:'follow' を指定して対応しています。
// ※ デプロイ設定：実行するユーザー＝自分、アクセスできるユーザー＝全員

  • 上記コードを貼り付けて保存（Ctrl+S）
  • デプロイ → 新しいデプロイ → ウェブアプリ
  • アクセス「全員」でデプロイ → URL をコピー
  • 下の欄に URL を貼り付け
AI カロリー解析（Gemini）

Google Gemini APIキーを設定すると、食事の写真からカロリーを自動推定できます。APIキーは端末内にのみ保存されます。

過去データ 一括取得

上のURLを設定した状態で「取得する」を押すと、スプレッドシートの1/5以降のデータをすべて自動でインポートします。

データリセット

⚠️ ローカルの全記録を削除します。スプシのデータは消えません。リセット後は「過去データを取得する」で再インポートしてください。
